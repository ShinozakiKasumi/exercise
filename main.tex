%%% 选择题环境（左栏题目 + 右栏答题卡）%%%
\NewDocumentEnvironment{choice}{m m}{%
	\par\addvspace{\medskipamount}%
	\stepcounter{choicecounter}%
	\stepcounter{answercnt}%
	\begin{tcolorbox}[
		enhanced,
		breakable,
		colback=lightbg,
		colframe=maincolor,
		boxrule=0.4pt,
		arc=2pt,
		left=5mm,
		right=3mm,
		top=4pt,
		bottom=4pt,
		boxsep=0pt,
		before upper={
			\noindent
			\textbf{\color{maincolor}习题 \arabic{choicecounter}}
			\hfill
			\knowledgepointicon~\text{#2} \quad
			{\color{accentcolor}[\ \textbf{#1}\ ]}
			\par\vspace{1ex}
		}
	]
}{%
	\end{tcolorbox}%
	\switchcolumn%
	%%% 右栏：答题卡 %%%
	\vspace{\medskipamount}%
	\stepcounter{answercnt}%
	\begin{tcolorbox}[
		enhanced,
		colback=answercolor,
		colframe=rulegray,
		boxrule=0.3pt,
		arc=1pt,
		left=4mm,
		right=3mm,
		top=4pt,
		bottom=4pt,
		boxsep=0pt,
		before upper={
			\noindent\textbf{\scriptsize\color{maincolor}答 \arabic{choicecounter}}\par\vspace{2pt}
			\scriptsize
		}
	]%
	\begin{tikzpicture}[
		optbox/.style={rectangle, draw=rulegray, minimum size=5mm, inner sep=0pt},
	]
		\foreach \l [count=\c] in {A,B,C,D} {
			\node[optbox] at (\c*7mm, 0) {};
			\node[font=\tiny] at (\c*7mm, -4mm) {\l};
		}
	\end{tikzpicture}%
	\end{tcolorbox}%
	\switchcolumn*%
}

%%% 非选择题环境（左栏题目 + 右栏作答横线）%%%
\NewDocumentEnvironment{exercise}{ O{} g g }{% O{}=>[note], g=>{score}, g=>{topic}
	\par\addvspace{\medskipamount}%
	\stepcounter{choicecounter}%
	\begin{tcolorbox}[
		enhanced,
		breakable,
		colback=lightbg,
		colframe=maincolor,
		boxrule=0.4pt,
		arc=2pt,
		left=5mm,
		right=3mm,
		top=4pt,
		bottom=4pt,
		boxsep=0pt,
		before upper={
			\noindent
			\textbf{\color{maincolor}习题 \thechoicecounter}
			\IfValueT{#1}{\ {\color{accentcolor}#1}}
			\IfValueT{#2}{\quad{\color{accentcolor}[本题 #2 分]}}
			\hfill
			\IfValueT{#3}{\knowledgepointicon~\text{#3}}
			\par\vspace{1ex}
		}
	]
}{%
	\end{tcolorbox}%
	\switchcolumn%
	%%% 右栏：作答区 %%%
	\vspace{\medskipamount}%
	\begin{tcolorbox}[
		enhanced,
		colback=answercolor,
		colframe=rulegray,
		boxrule=0.3pt,
		arc=1pt,
		left=4mm,
		right=3mm,
		top=4pt,
		bottom=4pt,
		boxsep=0pt,
		before upper={
			\noindent\textbf{\scriptsize\color{maincolor}答 \thechoicecounter}\par\vspace{2pt}
			\scriptsize
		}
	]%
	\color{rulegray}%
	\foreach \i in {1,...,3}{%
		\rule{\linewidth}{0.3pt}\par\vspace{8mm}%
	}%
	\end{tcolorbox}%
	\switchcolumn*%
}
